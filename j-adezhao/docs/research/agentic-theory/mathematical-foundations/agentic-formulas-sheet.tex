\section*{Agentic Formulas Sheet}

\subsection*{I. Assumptiones (Assumptions)}
\begin{itemize}
  \item All component scores are normalized to $[0,1]$ unless stated otherwise.
  \item Weight vectors satisfy simplex constraints: $\sum_i w_i = 1,\; w_i \ge 0$.
  \item Policy checks are binary for release gates: $0$ (fail), $1$ (pass).
\end{itemize}

\subsection*{II. Utility and Throughput}
\[
U = w_qQ - w_cC - w_rR + w_aA
\]
where $Q$ = quality, $C$ = normalized cost, $R$ = normalized risk, and $A$ = alignment to objective.

\[
G = \frac{T_h-T_a}{T_h}
\]
where $T_h$ is human-only completion time and $T_a$ is agent-assisted completion time.

\subsection*{III. Trust Scoring Model}
\[
T = w_1\,\text{Transparency}+w_2\,\text{Accuracy}+w_3\,\text{CulturalFit}
\]
with:
\[
w_1+w_2+w_3=1,\quad w_i\ge 0
\]

\subsection*{IV. Safe Action Gate}
\[
\text{ApproveAction}\iff (\text{Confidence}\ge\tau_c)\land(\text{Risk}\le\tau_r)\land(\text{PolicyCheck}=1)
\]

\subsection*{V. Override-Aware Trust Dynamics}
For step index $t$:
\[
T_{t+1} = \lambda T_t + (1-\lambda)\hat{T}_{t} - \delta \cdot \text{Override}_t
\]
where:
\begin{itemize}
  \item $\lambda \in [0,1]$ is persistence of prior trust,
  \item $\hat{T}_t$ is current interaction trust estimate from the model above,
  \item $\text{Override}_t \in \{0,1\}$ and $\delta > 0$ is override penalty.
\end{itemize}

\subsection*{VI. Madrid Pilot Worked Example}
Suppose one interaction has:
\[
\text{Transparency}=0.70,\;\text{Accuracy}=0.82,\;\text{CulturalFit}=0.64
\]
and weights:
\[
w_1=0.30,\;w_2=0.45,\;w_3=0.25
\]
Then:
\[
T = 0.30(0.70)+0.45(0.82)+0.25(0.64)=0.739
\]
If prior trust is $T_t=0.76$, with $\lambda=0.8$, $\delta=0.06$, and one override:
\[
T_{t+1}=0.8(0.76)+0.2(0.739)-0.06(1)=0.6958
\]
Interpretation: even with strong accuracy, avoidable overrides materially reduce next-step trust.

\subsection*{VII. Reporting Fields}
For each run, log:
\begin{itemize}
  \item $(Q,C,R,A,U)$
  \item $(\text{Transparency}, \text{Accuracy}, \text{CulturalFit}, T_t, T_{t+1})$
  \item gate outcome and override events
\end{itemize}
This keeps theory, UX evidence, and governance controls in one measurable pipeline.
