\section*{Study Tips}

\begin{enumerate}
    \item \textbf{Start Early}: Begin reviewing questions 2-3 weeks before exams
    \item \textbf{Active Practice}: Write out answers, don't just read them
    \item \textbf{Use Answer Keys}: Check your answers, then review incorrect ones
    \item \textbf{Create Flashcards}: Use fill-in-the-blank questions for flashcards
    \item \textbf{Group Study}: Discuss essay questions with classmates
    \item \textbf{Time Yourself}: Practice answering questions within time limits
\end{enumerate}
