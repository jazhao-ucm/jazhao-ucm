\footnotesize C o qu, pronunciaci\'on (Carlos, Garc\'ia, Javier, Celia), presente.
\end{center}\vspace{0.5em}

\section*{\color{w4brown}C o qu}
\textbf{qu} before e, i: qui\'en, qu\'e, quince, arquitectura, marroqu\'i. \\
\textbf{c} elsewhere: once, camarero, cocinero, cient\'ifica, gracias, comercial, cu\'antos, cuatro. \\
\textit{Exception:} \textbf{qui\'en} needs \textbf{qu}.

\section*{\color{w4brown}Pronunciaci\'on -- Reference words}
\textbf{Como Carlos:} C before a, o, u; Q = [k]. (colombiana, Costa, Quito.) \\
\textbf{Como Garc\'ia:} G before a, o, u = [g]. (Bogot\'a.) \\
\textbf{Como Javier:} J; G before e, i = [x]. (argentina, San Jos\'e.) \\
\textbf{Como Celia:} C before e, i; Z = [s] or [$\theta$]. (Rica, Tegucigalpa, Asunci\'on.)

\section*{\color{w4brown}Presente -- Verbos regulares}
\textbf{-ar:} hablo, hablas, habla, hablamos, habl\'ais, hablan. \\
\textbf{-er:} como, comes, come, comemos, com\'eis, comen. \\
\textbf{-ir:} vivo, vives, vive, vivimos, viv\'is, viven. \\
Irregulares: ser, estar, ir, tener, hacer.
