\section*{Four Blocks}\handnote{History, institutions, policies, foreign.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Block 1}: History, origins, Schuman, Rome.
\item \textbf{Block 2}: Institutions (exec, legis, judiciary, ECB, etc.).
\item \textbf{Block 3}: Policies (euro, identity, enlargement, migration). Most relevant.
\item \textbf{Block 4}: Foreign \& security. CFSP.
\end{itemize}

\section*{Key Concepts}\handnote{Two perspectives.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{\hlpink{Supranational}} = Above national. EU beyond single countries.
\item \textbf{\hlpink{Official story}} = EU narrates success, unity.
\item \textbf{\hlpink{Critical view}} = Realist. Contingency. Crises.
\item 27 member states (post-Brexit).
\end{itemize}

\section*{History -- Milestones}\handnote{Know the sequence.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Schuman Plan} (1950): Coal \& steel. Avoid tariffs.
\item \textbf{Treaty of Rome} (1957): EEC.
\item \textbf{Luxembourg Compromise}: National veto when interest at stake.
\item \textbf{Southern enlargement}: Greece (1981), Spain \& Portugal (1986).
\item \textbf{SEA}: Single market. Deeper integration.
\item \textbf{Maastricht} (1992): EU. Euro.
\item \textbf{Schengen}: No borders. Free movement.
\item UK opt-outs: no euro, some policies.
\end{itemize}

\section*{Southern Enlargement}\handnote{Democracy consolidation.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Greece (1981)}: Dictatorship fell 1974. EC = democracy consolidation.
\item \textbf{Spain \& Portugal (1986)}: Carnation Revolution (Portugal). Spain: transition after Franco.
\item Challenges: CAP, fishing, steel, textiles. Cohesion for poorer members.
\item \textbf{Cohesion}: Structural funds, EIB, Committee of Regions.
\end{itemize}

\section*{EU Institutions}\handnote{Exec, legis, judiciary.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Executive}: European Council + Commission.
\item \textbf{Legislative}: Parliament + Council of EU.
\item \textbf{Judiciary}: Court of Justice.
\item \textbf{Other}: ECB, Court of Auditors, agencies.
\end{itemize}

\section*{Key Readings}\handnote{Zweig, Judt, Anderson, Vinen.}
\begin{itemize}[leftmargin=*,nosep]
\item Zweig: Europe that disappeared. Borders, identity.
\item Judt: Origins. Anderson: Ever closer union. Vinen: Fragments.
\item Euroscepticism, Brexit, Anglosphere.
\end{itemize}

\section*{Quick revision}\handnote{Run through before exam.}
\begin{itemize}[leftmargin=*,nosep]
\item Four blocks. Supranational. Official vs.\ critical.
\item Schuman, Rome, Maastricht, Schengen.
\item Southern enlargement. Luxembourg Compromise.
\item Institutions. Cohesion. Readings.
\end{itemize}

\section*{\hlpeach{Text analysis tips}}\handnote{Analyse, don't summarise.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Main idea?} Central argument.
\item \textbf{What are they arguing?} Logic. Cause \& effect.
\item \textbf{Key expression?} Quote that captures it.
\item Compare official vs.\ critical. Connect to themes.
\end{itemize}
