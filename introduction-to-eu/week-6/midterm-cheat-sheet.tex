\section*{Unit 0 -- Course \& Assessment}\handnote{Analyse, don't summarise.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Midterm}: In-person. 2--3 open questions. Short texts. Analyse---main idea, argument, key expression. Not summarising.
\item \textbf{Essay}: 3,500 words. Topic agreed with professor. Balanced argument.
\item \textbf{Presentations}: Two readings per session. Debate. Participation valued.
\item EU in tense moment; history full of crises. Comparative angle: US, China, Russia.
\item \textbf{Final}: 19 May. Cumulative; emphasis on later units.
\end{itemize}

\section*{Unit 1 -- Four Blocks}\handnote{History, institutions, policies, foreign.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Block 1}: History, origins. Schuman, Rome. Economics and money. Logic and milestones, not just dates.
\item \textbf{Block 2}: Institutions. ``Even more complex.'' Executive, legislative, judiciary, other.
\item \textbf{Block 3}: Policies. Euro, identity, enlargement, migration. \textbf{Most relevant} for many.
\item \textbf{Block 4}: Foreign \& security. CFSP. Guest expert.
\end{itemize}

\section*{Unit 2 -- Key Concepts}\handnote{Two perspectives.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{\hlpink{Supranational}} = Above national. EU beyond single countries. Institutions with authority over states.
\item \textbf{\hlpink{Official story}} = EU narrates success, unity, integration. Pro-integration narrative.
\item \textbf{\hlpink{Critical view}} = Realist. Contingency. Crises. Not inevitable. Dissent, alternatives.
\item \textbf{\hlpink{Intergovernmentalism}} = States cooperate but keep veto. Sovereignty retained.
\item 27 member states (post-Brexit). Analytical tools: federation, government.
\item \textbf{Theory $\neq$ practice}: Reality is messy. Focus on practice.
\end{itemize}

\section*{Unit 3 -- History -- Milestones}\handnote{Know the sequence.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{1945--1954}: Marshall Plan, OEEC, Council of Europe (1949), NATO (1949), ECSC (1951), EDC failed (1954). France said no to European army (sovereignty).
\item \textbf{Schuman Plan} (9 May 1950): Coal \& steel under common High Authority. ``Europe Day.''
\item \textbf{Treaty of Paris} (1951): ECSC. First supranational transfer. Six: FR, DE, IT, BE, NL, LU.
\item \textbf{Treaty of Rome} (1957): EEC + Euratom. Customs union, common market.
\item \textbf{Empty chair crisis} (1965--66): De Gaulle withdrew France. Opposed own resources, QMV.
\item \textbf{Luxembourg Compromise} (1966): ``Vital national interest'' $\to$ veto. Paralysis $\sim$15 years.
\item \textbf{1967}: ECSC, EEC, Euratom merged $\to$ European Community (EC). Customs union 1968.
\item \textbf{Hague Summit} (1969): Pompidou. Enlargement, EMU. ``Bandwagon'' rolled again.
\item \textbf{1973}: UK, Ireland, Denmark join. Norway: referendum no (fishing). UK vetoed twice by de Gaulle (1963, 1967); joined after he left. EFTA = UK's alternative (free trade, no supranational).
\item \textbf{Oil crisis} (1973): OPEC weapon. Prices $\times$4. Stagflation. No common EEC response.
\item \textbf{Southern enlargement}: Greece (1981), Spain \& Portugal (1986).
\item \textbf{SEA} (1986): Single market. Qualified majority voting extended.
\item \textbf{Maastricht} (1992): EU, euro path, three pillars, citizenship. UK opt-outs.
\item \textbf{Schengen}: No internal borders. 26 countries, 400M+.
\item \textbf{Lisbon} (2007): Streamlined. More power to Parliament.
\end{itemize}\handnote{Schuman $\to$ Rome $\to$ Luxembourg $\to$ Southern $\to$ Maastricht.}

\subsection*{Four freedoms \& Copenhagen criteria}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Four freedoms}: Goods, services, capital, people. Free movement across borders.
\item \textbf{Copenhagen}: Democracy, rule of law, human rights; market economy; adopt \textit{acquis}.
\end{itemize}

\subsection*{Enlargements timeline}
\begin{itemize}[leftmargin=*,nosep]
\item 1973: UK, Ireland, Denmark (Norway: no). 1981: Greece. 1986: Spain, Portugal.
\item 1995: Austria, Finland, Sweden. 2004: Big Bang (10 East). 2007: Bulgaria, Romania. 2013: Croatia.
\end{itemize}

\subsection*{CAP \& oil crisis}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{CAP} (1962): French priority. Common prices, subsidies. Butter mountains, wine lakes. Still controversial.
\item \textbf{Oil 1973}: OPEC weapon. Stagflation. No common EEC energy policy. Nuclear (France), diversification.
\end{itemize}

\section*{Unit 4 -- Southern Enlargement}\handnote{Democracy consolidation.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Greece (1981)}: Colonels 1967--74. Cyprus conflict. EC = democracy consolidation.
\item \textbf{Portugal (1986)}: Carnation Revolution 1974. Decolonisation. Revolution, not negotiated.
\item \textbf{Spain (1986)}: Franco died 1975. Negotiated transition. 1982 World Cup = soft power.
\item Challenges: France worried about CAP (large farming). Fishing, steel, textiles. Cohesion for poorer.
\item \textbf{Cohesion policy}: Structural funds, EIB, Committee of Regions. Redistribution.
\item External (US, fear of communism) and internal both important.
\item \textbf{Gibraltar}: Border closed 1969 (Franco). Reopened 1982. Part of Spain's accession talks.
\end{itemize}

\section*{Unit 5 -- EU Institutions}\handnote{Exec, legis, judiciary.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Executive}: European Council (heads of state) + Commission (proposes, implements). ``Twin executive.''
\item \textbf{Legislative}: Parliament (elected, co-decision) + Council of EU (ministers).
\item \textbf{Judiciary}: Court of Justice. Supremacy of EU law (established 1960s--70s).
\item \textbf{Other}: ECB, Court of Auditors, agencies (EUIPO, etc.).
\item Commission proposes; Council decides. France + Germany = ``engine.''
\end{itemize}

\section*{Unit 6 -- Key Readings}\handnote{Zweig, Judt, Krawatzek \& Pestel, Anderson, Vinen, Bijsmans, Gilbert, Kenny \& Pearce, Delors.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Zweig} (\textit{World of Yesterday}): Europe that disappeared. Borders, xenophobia, bureaucracy after WWI. Freedom vs.\ security.
\item \textbf{Judt}: British reluctance. Identity (Commonwealth, ``we won''), power, institutions. Messy integration.
\item \textbf{Krawatzek \& Pestel}: Top-down bias. Pro-European bias. Need diverse, vernacular voices.
\item \textbf{Anderson}: Democratic deficit. ECJ, unelected institutions. ``Ever closer union?'' (question mark).
\item \textbf{Vinen}: History in fragments. 1989 did not ``unite'' Europe. Many Europes. Class, region, age.
\item \textbf{Bijsmans}: Euroscepticism multifaceted. Hard vs.\ soft. Economic, identity, democratic drivers.
\item \textbf{Gilbert}: Teleology as problem. Contingency, dissent, losers. Avoid ``success story'' histories.
\item \textbf{Kenny \& Pearce}: Brexit \& Anglosphere. British identity: Europe vs.\ English-speaking world. Imagined community.
\item \textbf{Delors}: Commission President 1985--95. SEA, Maastricht, Schengen, cohesion.
\end{itemize}

\subsection*{Historiography \& teleology}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Teleology} = History as destiny. ``Had to happen.'' Avoid.
\item \textbf{Contingency}: Actors didn't know outcome. Failure possible. Alternatives existed.
\item \textbf{Histories ``for'' Europe}: Pro-integration, elite-centred. Blank spaces: dissent, colonialism.
\end{itemize}

\subsection*{Euroscepticism (Bijsmans)}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Hard}: Reject EU. Withdrawal. \textbf{Soft}: Accept membership, criticise policies.
\item \textbf{Permissive consensus} $\to$ \textbf{constraining dissensus} (Maastricht).
\item Drivers: Economic (winners/losers), identity, democratic deficit.
\item Constructive (reform, facts) vs.\ destructive (delegitimise, scapegoats).
\end{itemize}

\section*{\hlpeach{Text analysis tips}}\handnote{Analyse, don't summarise.}
\begin{itemize}[leftmargin=*,nosep]
\item \textbf{Main idea?} Central argument. Don't summarise everything.
\item \textbf{What are they arguing?} Logic. Cause \& effect.
\item \textbf{Key expression?} Quote that captures the argument.
\item Compare official vs.\ critical. Connect to course themes.
\item Focus on understanding and critical reading, not memorisation of dates.
\end{itemize}
