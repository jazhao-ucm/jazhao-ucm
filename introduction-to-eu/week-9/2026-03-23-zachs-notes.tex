\section*{VII. Extra institutional recap (from transcript) --- Democratic deficit}
\begin{enumerate}[label=\Alph*.]
\item \textbf{Democratic deficit (core concept)}
  \begin{enumerate}[label=\arabic*.]
  \item Gap: EU decision-making vs citizens' influence/control
  \item If citizens cannot control ``how much influences citizens have'', then democratic deficit
  \end{enumerate}
\item \textbf{Early integration dynamic (as described)}
  \begin{enumerate}[label=\arabic*.]
  \item Decisions made largely via national contexts + officials (esp. within Council)
  \item Commission/Council had strong role; EP supposed to represent citizens but limited power
  \end{enumerate}
\item \textbf{Why it matters for today's debate lens}
  \begin{enumerate}[label=\arabic*.]
  \item Connects ``institutions + who controls law-making'' to workshop tasks (parliament vocabulary, procedures, groups)
  \end{enumerate}
\end{enumerate}
